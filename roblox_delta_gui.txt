--==================================================
-- DRAG + CLICK SYSTEM
--==================================================
local dragging = false
local moved = false
local dragStart
local startPos
local guiAberta = false

floatButton.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or 
       input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        moved = false
        dragStart = input.Position
        startPos = floatButton.Position
    end
end)

floatButton.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or 
       input.UserInputType == Enum.UserInputType.Touch then
        dragging = false
        
        if not moved then
            guiAberta = not guiAberta
            
            if guiAberta then
                mainFrame.Visible = true
                mainFrame.Size = UDim2.fromOffset(0, 0)
                mainFrame.Position = UDim2.new(0.5, 0, 0.5, 0)
                
                TweenService:Create(
                    mainFrame,
                    TweenInfo.new(0.4, Enum.EasingStyle.Back, Enum.EasingDirection.Out),
                    {
                        Size = UDim2.fromOffset(620, 480),
                        Position = UDim2.new(0.5, -310, 0.5, -240)
                    }
                ):Play()
            else
                local tween = TweenService:Create(
                    mainFrame,
                    TweenInfo.new(0.3, Enum.EasingStyle.Back, Enum.EasingDirection.In),
                    {
                        Size = UDim2.fromOffset(0, 0),
                        Position = UDim2.new(0.5, 0, 0.5, 0)
                    }
                )
                tween:Play()
                tween.Completed:Connect(function()
                    mainFrame.Visible = false
                end)
            end
        end
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or 
                     input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - dragStart
        
        if math.abs(delta.X) > CLICK_DISTANCE or math.abs(delta.Y) > CLICK_DISTANCE then
            moved = true
        end
        
        floatButton.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end
end)

-- Close button functionality
closeButton.MouseButton1Click:Connect(function()
    guiAberta = false

        }
    )
    tween:Play()
    tween.Completed:Connect(function()
        mainFrame.Visible = false
    end)
end)

local tabs = {}
local currentTab
local selectedButton

local function createTab(name, icon)
    -- Bot√£o da aba
    local tabBtn = Instance.new("TextButton", sidebar)
    tabBtn.Size = UDim2.new(1, -12, 0, 42)
    tabBtn.Text = (icon or "‚òÖ") .. " " .. name
    tabBtn.Font = Enum.Font.GothamBold
    tabBtn.TextSize = 13
    tabBtn.TextColor3 = Color3.fromRGB(200, 200, 200)
    tabBtn.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
    tabBtn.AutoButtonColor = false
    tabBtn.ZIndex = 7
    
    local btnCorner = Instance.new("UICorner", tabBtn)
    btnCorner.CornerRadius = UDim.new(0, 10)
    
    local grad = Instance.new("UIGradient", tabBtn)
    grad.Color = ColorSequence.new{
        ColorSequenceKeypoint.new(0, Color3.fromRGB(0, 170, 255)),
        ColorSequenceKeypoint.new(1, Color3.fromRGB(170, 0, 255))
    }
    grad.Transparency = NumberSequence.new(0.7)
    
    -- ScrollingFrame para o conte√∫do da aba
    local tabFrame = Instance.new("ScrollingFrame", content)
    tabFrame.Size = UDim2.new(1, 0, 1, 0)
    tabFrame.Position = UDim2.new(1, 0, 0, 0)
    tabFrame.Visible = false
    tabFrame.BackgroundTransparency = 1
    tabFrame.BorderSizePixel = 0
    tabFrame.ScrollBarThickness = 6
    tabFrame.ScrollBarImageColor3 = Color3.fromRGB(80, 80, 80)
    tabFrame.ZIndex = 6
    
    local tabLayout = Instance.new("UIListLayout", tabFrame)
    tabLayout.Padding = UDim.new(0, 8)
    tabLayout.HorizontalAlignment = Enum.HorizontalAlignment.Left
    
    local tabPadding = Instance.new("UIPadding", tabFrame)
    tabPadding.PaddingTop = UDim.new(0, 10)
    tabPadding.PaddingLeft = UDim.new(0, 10)
    tabPadding.PaddingRight = UDim.new(0, 10)
    
    tabLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
        tabFrame.CanvasSize = UDim2.new(0, 0, 0, tabLayout.AbsoluteContentSize.Y + 20)
    end)
    
    -- Hover effect
    tabBtn.MouseEnter:Connect(function()
        if tabBtn ~= selectedButton then
            TweenService:Create(tabBtn, TweenInfo.new(0.2), {
                TextColor3 = Color3.new(1, 1, 1),
                BackgroundColor3 = Color3.fromRGB(35, 35, 35)
            }):Play()
        end
    end)
    
    tabBtn.MouseLeave:Connect(function()
        if tabBtn ~= selectedButton then
            TweenService:Create(tabBtn, TweenInfo.new(0.2), {
                TextColor3 = Color3.fromRGB(200, 200, 200),
                BackgroundColor3 = Color3.fromRGB(30, 30, 30)
            }):Play()
        end
    end)
    
    -- Click handler
    tabBtn.MouseButton1Click:Connect(function()
        if currentTab then
            currentTab.Visible = false
        end
        
        if selectedButton then
            TweenService:Create(selectedButton, TweenInfo.new(0.2), {
                TextColor3 = Color3.fromRGB(200, 200, 200),
                BackgroundColor3 = Color3.fromRGB(30, 30, 30)
            }):Play()
            local oldGrad = selectedButton:FindFirstChildOfClass("UIGradient")
            if oldGrad then
                TweenService:Create(oldGrad, TweenInfo.new(0.2), {
                    Transparency = NumberSequence.new(0.7)
                }):Play()
            end
        end
        
        tabFrame.Position = UDim2.new(1, 0, 0, 0)
        tabFrame.Visible = true
        
        TweenService:Create(tabFrame, TweenInfo.new(0.3, Enum.EasingStyle.Quint), {
            Position = UDim2.new(0, 0, 0, 0)
        }):Play()
        
        TweenService:Create(tabBtn, TweenInfo.new(0.2), {
            TextColor3 = Color3.new(1, 1, 1),
            BackgroundColor3 = Color3.fromRGB(40, 40, 40)
        }):Play()
        
        TweenService:Create(grad, TweenInfo.new(0.2), {
            Transparency = NumberSequence.new(0.3)
        }):Play()
        
        currentTab = tabFrame
        selectedButton = tabBtn
    end)
    
    tabs[name] = tabFrame
    return tabFrame
end

    switchCircle.Size = UDim2.fromOffset(14, 14)
    switchCircle.Position = UDim2.new(0, 2, 0.5, -7)
    switchCircle.BackgroundColor3 = Color3.new(1, 1, 1)
    switchCircle.ZIndex = 9
    
    Instance.new("UICorner", switchCircle).CornerRadius = UDim.new(1, 0)
    
    local on = false
    
    local clickDetector = Instance.new("TextButton", toggle)
    clickDetector.Size = UDim2.new(1, 0, 1, 0)
    clickDetector.BackgroundTransparency = 1
    clickDetector.Text = ""
    clickDetector.ZIndex = 10
    
    clickDetector.MouseButton1Click:Connect(function()
        on = not on
        
        TweenService:Create(switchBg, TweenInfo.new(0.25), {
            BackgroundColor3 = on and Color3.fromRGB(0, 200, 100) or Color3.fromRGB(150, 0, 0)
        }):Play()
        
        TweenService:Create(switchCircle, TweenInfo.new(0.25, Enum.EasingStyle.Quint), {
            Position = on and UDim2.new(1, -16, 0.5, -7) or UDim2.new(0, 2, 0.5, -7)
        }):Play()
        
        if callback then
            pcall(function()
                callback(on)
            end)
        end
    end)
    
    return toggle
end

local function createButton(parent, text, callback)
    local button = Instance.new("TextButton", parent)
    button.Size = UDim2.new(1, -10, 0, 38)
    button.Text = text
    button.Font = Enum.Font.GothamBold
    button.TextSize = 13
    button.TextColor3 = Color3.new(1, 1, 1)
    button.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
    button.AutoButtonColor = false
    button.ZIndex = 7
    
    Instance.new("UICorner", button).CornerRadius = UDim.new(0, 10)
    
    local grad = Instance.new("UIGradient", button)
    grad.Color = ColorSequence.new{
        ColorSequenceKeypoint.new(0, Color3.fromRGB(0, 200, 255)),
        ColorSequenceKeypoint.new(1, Color3.fromRGB(200, 0, 255))
    }
    grad.Transparency = NumberSequence.new(0.6)
    
    button.MouseEnter:Connect(function()
        TweenService:Create(grad, TweenInfo.new(0.2), {
            Transparency = NumberSequence.new(0.3)
        }):Play()
        TweenService:Create(button, TweenInfo.new(0.2), {
            BackgroundColor3 = Color3.fromRGB(50, 50, 50)
        }):Play()
    end)
    
    button.MouseLeave:Connect(function()
        TweenService:Create(grad, TweenInfo.new(0.2), {
            Transparency = NumberSequence.new(0.6)
        }):Play()
        TweenService:Create(button, TweenInfo.new(0.2), {
            BackgroundColor3 = Color3.fromRGB(40, 40, 40)
        }):Play()
    end)
    
    if callback then
        button.MouseButton1Click:Connect(function()
            pcall(callback)
        end)
    end
    
    return button
end
